async function fetchData() { const selectedType = document.getElementById('typeFilter').value; const dateVal = document.getElementById('dateFilter').value; const status = document.getElementById('statusText'); allData = []; document.getElementById('dataBody').innerHTML = ""; // Muunnetaan päivämäärä ISO-muotoon (esim. 2024-03-20T00:00:00.000Z) // Rajapinta tarvitsee tämän nimenomaan createdOrUpdatedAfter -parametrille const createdOrUpdatedAfter = dateVal ? new Date(dateVal).toISOString() : ""; let offset = 0; let hasMore = true; try { while (hasMore) { status.innerText = `Haetaan... (Löytynyt: ${allData.length})`; // Käytetään nyt täsmälleen oikeaa parametrin nimeä: createdOrUpdatedAfter let apiUrl = `https://www.lahella.fi/v1/integration/activities?type=${selectedType}&offset=${offset}`; if (createdOrUpdatedAfter) { apiUrl += `&createdOrUpdatedAfter=${encodeURIComponent(createdOrUpdatedAfter)}`; } const res = await fetch(PROXY + encodeURIComponent(apiUrl)); const json = await res.json(); const items = json.items || []; if (items.length === 0) break; items.forEach(item => { const traits = item.traits || {}; // SELAINPUOLEN VARMISTUS: // Varmistetaan vielä koodissa, että kategoria ja päivämäärä täsmäävät, // jos rajapinta sattuisi palauttamaan liikaa dataa. const itemTypes = traits.type || []; const typeMatch = itemTypes.includes(selectedType); const itemDate = new Date(item.updatedAt || item.createdAt); const filterDate = dateVal ? new Date(dateVal) : null; const dateMatch = !filterDate || itemDate >= filterDate; if (typeMatch && dateMatch) { let ways = new Set(); let codes = new Set(); let cities = new Set(); (traits.channels || []).forEach(ch => { if (ch.type) ch.type.forEach(t => ways.add(t)); if (ch.address?.postalCode) codes.add(ch.address.postalCode); if (ch.address?.city) cities.add(ch.address.city); }); allData.push({ "Nimi": traits.name || "Nimetön", "Järjestö-ID": item.group, "Osallistumistapa": Array.from(ways).join(", "), "Postinumerot": Array.from(codes).join(", "), "Kaupunki": Array.from(cities).join(", "), "Päivitetty": itemDate.toLocaleDateString('fi-FI'), "Linkki": `https://www.lahella.fi/activities/${item._key}` }); } }); renderTable(); offset += items.length; // Jos hasMore on false tai ollaan kahlattu jo kauas menneisyyteen, lopetetaan hasMore = json.hasMore === true && offset < 5000; // Lisäjarru: Jos uusin saatu erä on kokonaan vanhempaa kuin hakupäivä, lopetetaan haku if (filterDate && items.length > 0) { const oldestInBatch = new Date(items[items.length - 1].updatedAt || items[items.length - 1].createdAt); if (oldestInBatch < filterDate) { hasMore = false; } } } status.innerText = `Haku valmis! Tuloksia: ${allData.length}`; } catch (e) { status.innerText = "Virhe: " + e.message; } }
